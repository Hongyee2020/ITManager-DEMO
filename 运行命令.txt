 # 启动脚本
 python manage.py runserver 0.0.0.0:8000

python manage.py makemigrations

python manage.py migrate

python manage.py sqlmigrate cmdb 0001

python manage.py sqlmigrate cmdb 0003

4. 检查模型
python manage.py check
